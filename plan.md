// ASSUME YOU HAVE: originalSlipCanvas (the messy user-pasted slip)
// ASSUME YOU HAVE: myNewFrameCanvas (the one that will look like image_0.png)
// And a dictionary mapping coordinates to field names

const frameCtx = myNewFrameCanvas.getContext('2d');

// --- STEP 1: PRE-RENDER THE "FRAME" PLATE ---
// This part is static and fast. You can do this once and reuse it.

function createBasePlate() {
  const c = frameCtx;
  const w = 600; // Define your fixed slip width
  const h = 700; // Define your fixed slip height

  // 1. Draw Background & Outer Grey Border
  c.fillStyle = '#f8f8f8'; // Base light grey border color
  c.fillRect(0, 0, w, h);
  c.fillStyle = '#ffffff'; // Interior white background
  c.fillRect(10, 10, w - 20, h - 20); // Create an inset border effect

  // 2. Draw Top Left Logo
  // Assuming 'SafeGridLogoImage' is pre-loaded
  c.drawImage(SafeGridLogoImage, 30, 40, 120, 40); // Standard positioning

  // 3. Draw Center Icon
  // Assuming 'SuccessCircleIcon' is pre-loaded
  c.drawImage(SuccessCircleIcon, w / 2 - 40, 110, 80, 80);

  // 4. Draw Success Text
  c.font = 'bold 24px sans-serif';
  c.fillStyle = '#0f9a56'; // Target Green color
  c.textAlign = 'center';
  c.fillText('SUCCESS', w / 2, 220);

  // 5. Draw Horizontal Dividers
  c.fillStyle = '#cccccc'; // Grey line color
  c.fillRect(30, 260, w - 60, 2); // Top Line
  c.fillRect(30, 440, w - 60, 2); // Bottom Line

  // 6. Draw Field Labels (Text)
  c.font = '18px sans-serif';
  c.fillStyle = '#333333';
  c.textAlign = 'right'; // We will align labels to the right

  // Define positioning for labels (adjust as needed)
  c.fillText('Name:', 150, 310);
  c.fillText('IFSC:', 150, 340);
  c.fillText('Account:', 150, 370);
  // Repeat for OrderNo, Date, UTR below the bottom divider
}

// --- STEP 2: THE "RE-RENDERING" MAGIC ---

function processImageFile(originalSlipCanvas) {
  // Use the pre-rendered base plate we created above
  createBasePlate(); // Draws SAFEGRID, SUCCESS, Labels onto myNewFrameCanvas

  const frameCtx = myNewFrameCanvas.getContext('2d');
  const w = myNewFrameCanvas.width;

  // Now we *selectively inject* ONLY the data we need.
  // We use coordinates from your "custom config" (where you used to mask).

  const myMaskConfig = {
    amount: { x: 50, y: 100, w: 200, h: 50 }, // Coordinates on the messy original
    name: { x: 50, y: 160, w: 200, h: 30 },
    // ... define other fields ...
  };

  frameCtx.font = '22px sans-serif';
  frameCtx.fillStyle = '#000000';
  frameCtx.textAlign = 'left';

  // **THE CRITICAL STEP:** We "cut" data and "paste" it into our new layout.
  // Amount (needs unique positioning)
  const amountSource = myMaskConfig.amount;
  frameCtx.textAlign = 'center'; // Center the big amount
  frameCtx.font = 'bold 36px sans-serif';
  frameCtx.drawImage(originalSlipCanvas, amountSource.x, amountSource.y, amountSource.w, amountSource.h, w / 2 - amountSource.w / 2, 170, amountSource.w, amountSource.h);

  // Name (standard position)
  const nameSource = myMaskConfig.name;
  frameCtx.textAlign = 'left'; // Align the name to the left
  frameCtx.font = '22px sans-serif';
  frameCtx.drawImage(originalSlipCanvas, nameSource.x, nameSource.y, nameSource.w, nameSource.h, 170, 290, nameSource.w, nameSource.h);

  // ... Repeat for IFSC, Account, OrderNo, Date, UTR ...

  // 3. DONE! The `myNewFrameCanvas` now contains the final output image.
  // You can pass THIS canvas result directly to your collage generator.
}
